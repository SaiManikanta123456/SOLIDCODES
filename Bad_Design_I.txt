class User {

    public void viewContent() {
        System.out.println("Viewing content");
    }

    public void deleteContent() {
        System.out.println("Deleting content");
    }
}


class GuestUser extends User {

    @Override
    public void deleteContent() {
        throw new UnsupportedOperationException("Guest cannot delete");
    }
}


public class Main {
    public static void main(String[] args) {

        User user = new GuestUser();
        user.deleteContent(); // 💥 runtime failure
    }
}
