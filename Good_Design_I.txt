interface User {
    void viewContent();
}


interface DeletableUser extends User {
    void deleteContent();
}


class AdminUser implements DeletableUser {

    public void viewContent() {
        System.out.println("Admin viewing content");
    }

    public void deleteContent() {
        System.out.println("Admin deleting content");
    }
}


class CustomerUser implements User {

    public void viewContent() {
        System.out.println("Customer viewing content");
    }
}


class GuestUser implements User {

    public void viewContent() {
        System.out.println("Guest viewing content");
    }
}


class ContentService {

    public void showContent(User user) {
        user.viewContent();
    }

    public void deleteContent(DeletableUser user) {
        user.deleteContent();
    }
}


public class Main {

    public static void main(String[] args) {

        ContentService service = new ContentService();

        User admin = new AdminUser();
        User customer = new CustomerUser();
        User guest = new GuestUser();

        // All users can view → works fine
        service.showContent(admin);
        service.showContent(customer);
        service.showContent(guest);

        // Only deletable users allowed
        DeletableUser deletable = new AdminUser();
        service.deleteContent(deletable);

        // ❌ Compile-time safety (VERY IMPORTANT)
        // service.deleteContent(new GuestUser()); // not allowed
    }
}
